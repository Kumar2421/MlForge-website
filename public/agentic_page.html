import mlforge as forge

kernel = forge.Kernel(api_key="forge_api_...")

    name="Object-Detection-v1",
    model="yolov10n.pt",
    config={"iou": 0.45, "conf": 0.25},
    device="cuda:0"
)

# Execute protocol and synchronize local state
pipeline.run_and_evaluate(kernel=kernel)
print(f"mAP: {pipeline.metrics.mAP}")
